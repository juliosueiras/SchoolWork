import java.util.Scanner;

/**  

* Here is a sample app class to reverse-print an input string 

* @author Matt Swan

*/

 

public class ReverseStringPrint {    

private static Scanner inText = new Scanner(System.in);

    public static void main(String[] args) {        

String sReverseText = "";     

System.out.println("Please type in a sentence.");        

String sText = inText.nextLine();        

for(int i=sText.length(); i != 0; i--)            

sReverseText += sText.substring(i-1,i);  

System.out.println("Result: " + sReverseText);    

}    

}
